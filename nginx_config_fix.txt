# Updated Nginx Configuration for mtumrah.com
# Copy this to /etc/nginx/sites-available/mtumrah

# Redirect non-www → www (keep POST/PUT)
server {
    listen 443 ssl;
    server_name mtumrah.com;

    ssl_certificate     /etc/letsencrypt/live/mtumrah.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/mtumrah.com/privkey.pem;
    include             /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam         /etc/letsencrypt/ssl-dhparams.pem;

    # IMPORTANT: Add this to handle large uploads during redirect
    client_max_body_size 100M;

    return 308 https://www.mtumrah.com$request_uri;
}

# HTTP → HTTPS
server {
    listen 80;
    server_name mtumrah.com www.mtumrah.com;
    
    # IMPORTANT: Add this to handle large uploads during redirect
    client_max_body_size 100M;
    
    return 301 https://www.mtumrah.com$request_uri;
}

# Main site
server {
    listen 443 ssl;
    server_name www.mtumrah.com;

    ssl_certificate     /etc/letsencrypt/live/mtumrah.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/mtumrah.com/privkey.pem;
    include             /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam         /etc/letsencrypt/ssl-dhparams.pem;

    client_max_body_size 100M;
    
    # Increase proxy buffers for large file uploads
    proxy_buffer_size 128k;
    proxy_buffers 4 256k;
    proxy_busy_buffers_size 256k;
    proxy_connect_timeout 300;
    proxy_send_timeout 300;
    proxy_read_timeout 300;
    send_timeout 300;

    # --- Backend (Laravel API) - MUST come BEFORE / location ---
    location /api {
        proxy_pass http://127.0.0.1:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header X-Forwarded-Host $host;

        # CORS headers - MUST use 'always' so they appear even on error responses
        add_header Access-Control-Allow-Origin "https://www.mtumrah.com" always;
        add_header Access-Control-Allow-Methods "GET, POST, PUT, PATCH, DELETE, OPTIONS" always;
        add_header Access-Control-Allow-Headers "Content-Type, Authorization, X-Requested-With" always;
        add_header Access-Control-Allow-Credentials "true" always;

        # Pre-flight OPTIONS requests
        if ($request_method = OPTIONS) {
            add_header Access-Control-Allow-Origin "https://www.mtumrah.com" always;
            add_header Access-Control-Allow-Methods "GET, POST, PUT, PATCH, DELETE, OPTIONS" always;
            add_header Access-Control-Allow-Headers "Content-Type, Authorization, X-Requested-With" always;
            add_header Access-Control-Allow-Credentials "true" always;
            add_header Content-Length 0;
            add_header Content-Type text/plain;
            return 204;
        }

        # Disable redirects from Laravel - force it to return JSON
        proxy_redirect off;
        
        # Large file upload support
        client_max_body_size 100M;
    }

    # --- Frontend (Next.js) - Must come AFTER /api ---
    location / {
        proxy_pass http://127.0.0.1:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }
}
